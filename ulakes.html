from scipy.signal import blackman, wiener, convolve, find_peaks_cwt
from pique import peak

window = blackman( 100 )
window = window / sum(window)

maxsnps = 0
for n,c in enumerate(bigcontigs) :
    for s in vcf_reader.samples :
        data = wiener( mysamples[s][c]['allvars'], mysize=50 )
        data = convolve( data, window, mode='same' )
        if max(mysamples[s][c]['allvars']) > maxsnps : maxsnps = max(mysamples[s][c]['allvars'])
        maxtab, mintab = peak.peakdet(data, 0.5)
        #maxtab = find_peaks_cwt( mysamples[s][c]['snps'], arange(10,100) )
        mysamples[s][c]['wiener'] = data
        mysamples[s][c]['maxima'] = maxtab
        mysamples[s][c]['minima'] = mintab
        mysamples[s][c]['total']  = sum(mysamples[s][c]['allvars'])
